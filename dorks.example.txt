# One dork (search query) per line. Blank lines and lines starting with # are ignored.
#
# A "dork" is just a search query that uses operators to narrow results. Each
# result URL becomes a seed page for the crawl.
#
# Provider matters for operators:
#   browser (Bing) : site: inurl: intitle: filetype: "exact phrase" -exclude   <- best keyless
#   brave          : site: intitle: inurl: ext: "exact phrase" -exclude
#   duckduckgo     : site: inurl: "exact phrase"  (weak at filetype:/intitle:, rate-limits fast)
#
# Broad dorks below have NO site: filter, so they surface links from many
# different domains. Run them with:  --search-provider browser
#
# Use these only against targets you are authorized to assess.

# --- Open directory listings (mixed domains) ---
intitle:index.of "parent directory"
intitle:index.of filetype:pdf
intitle:index.of mp3

# --- Documents across the web ---
filetype:pdf "annual report"
filetype:xlsx "email"
filetype:log

# --- Common app / CMS fingerprints ---
inurl:wp-content
inurl:phpinfo.php
inurl:admin login

# --- Exact-phrase discovery ---
"powered by" forum
"index of /backup"

# --- Site-scoped examples (work on any provider) ---
site:python.org inurl:downloads
site:europa.eu filetype:pdf
