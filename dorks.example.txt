# One dork (search query) per line. Blank lines and lines starting with # are ignored.
#
# A "dork" is just a search query that uses operators to narrow results. These
# are fed to the configured --search-provider, and every result URL becomes a
# seed page for the crawl.
#
# Operator support depends on the provider:
#   duckduckgo : site:  inurl:  intitle:  filetype:  "exact phrase"  -exclude
#   searxng    : depends on the engines the instance enables (usually all)
#   brave      : site:  intitle:  inurl:  ext:  "exact phrase"  -exclude
#   google     : full Google operator set (within the configured search engine)
#
# Examples (edit these for your own authorized targets):

# All indexed pages on one host
site:python.org

# Narrow a site to a path/keyword
site:python.org inurl:downloads

# A specific document type on a domain
site:europa.eu filetype:pdf

# Title match
intitle:"release notes" site:djangoproject.com

# Exact phrase plus an exclusion
"getting started" site:fastapi.tiangolo.com -inurl:blog

# Combine hosts with OR
site:docs.python.org OR site:peps.python.org
